                        list1=[[func1,'argument for func1'],[func2,'argument for func2'],[func3,'argument for func3',]]


                        # access functions through the list of lists [[functionHere,argumentHere],[functionHere,argumentHere]]

                        # Dictionary version: note you can not have duplicate keys in a dict which will be a problem when sequencing function calls
                        funcDict={func1: "func1",func2: "func2",func3: "func3"}

                        print(funcDict)

                        print('dictionary calls here:')
                        for k,v in funcDict.items():
                        k(v)
